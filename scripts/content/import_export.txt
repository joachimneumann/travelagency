# Content import/export runbook

Local repo:
cd /Users/joachim/projects/travelagency

ATP staging repo:
cd /srv/asiatravelplan-staging

Remote SSH alias:
atp


## compare local content vs ATP

./scripts/content/compare_content.sh


## export local content to ATP

rsync -av content/atp_staff/ atp:/srv/asiatravelplan-staging/content/atp_staff/
rsync -av --delete content/tours/ atp:/srv/asiatravelplan-staging/content/tours/
rsync -av --delete content/translations/ atp:/srv/asiatravelplan-staging/content/translations/


## import ATP content to local

rsync -av atp:/srv/asiatravelplan-staging/content/atp_staff/ content/atp_staff/
rsync -av --delete atp:/srv/asiatravelplan-staging/content/tours/ content/tours/
rsync -av --delete atp:/srv/asiatravelplan-staging/content/translations/ content/translations/


## commit translations to git

content/translations is ignored by content/**, so force-add it when a translation snapshot must be portable through git. Human-maintained translation policy lives in config/i18n and is committed normally.

git status --short content/translations config/i18n scripts/i18n/source_catalogs
git add -f content/translations
git add config/i18n
git add scripts/i18n/source_catalogs
git commit -m "Update translation snapshots"


## restore translations from git

Restore the committed translation snapshots into the working tree.

git restore content/translations
git restore config/i18n
git restore scripts/i18n/source_catalogs

If content/translations was deleted locally and git does not restore ignored paths in your shell, use checkout:

git checkout HEAD -- content/translations config/i18n scripts/i18n/source_catalogs


## transfer translations through git: local to ATP

On local:

git add -f content/translations
git add config/i18n
git add scripts/i18n/source_catalogs
git commit -m "Update translation snapshots"
git push origin main

On ATP:

git pull --ff-only
node backend/app/scripts/backfill_marketing_tour_translations.js --validate


## transfer translations through git: ATP to local

On ATP:

git add -f content/translations
git add config/i18n
git add scripts/i18n/source_catalogs
git commit -m "Update translation snapshots from ATP"
git push origin main

On local:

git pull --ff-only
node backend/app/scripts/backfill_marketing_tour_translations.js --validate


## transfer translations directly: local to ATP

rsync -av --delete content/translations/ atp:/srv/asiatravelplan-staging/content/translations/
rsync -av scripts/i18n/source_catalogs/ atp:/srv/asiatravelplan-staging/scripts/i18n/source_catalogs/


## transfer translations directly: ATP to local

rsync -av --delete atp:/srv/asiatravelplan-staging/content/translations/ content/translations/
rsync -av atp:/srv/asiatravelplan-staging/scripts/i18n/source_catalogs/ scripts/i18n/source_catalogs/


## validate translations

node backend/app/scripts/backfill_marketing_tour_translations.js --validate
node scripts/i18n/build_runtime_i18n.mjs --strict


## copy content to backup

rsync -av content/atp_staff/ backup/atp_staff/
rsync -av --delete content/tours/ backup/tours/
rsync -av --delete content/translations/ backup/translations/


## restore content from backup

rsync -av backup/atp_staff/ content/atp_staff/
rsync -av --delete backup/tours/ content/tours/
rsync -av --delete backup/translations/ content/translations/
